#include<stdio.h>
int main() {

    int gross_salary,allowances,deductions,net_salary;

    printf("Enter your Gross Salary = ");
    scanf("%d",& gross_salary);

    if (gross_salary > 10000) {

        int allowances = gross_salary * 10 / 100;
        int deductions = gross_salary * 3 / 100;
        int net_salary = gross_salary + allowances - deductions;
        printf("Your Net Salary is  = %d",net_salary);

    }
    else  if (gross_salary > 5000) {

        int allowances = gross_salary * 7/ 100;
        int deductions = gross_salary * 2 / 100;
        int net_salary = gross_salary + allowances - deductions;
        printf("Your Net Salary is  =%d",net_salary);
    }
 
    return 0;

}
